# VB6/VB 中 MSMQ 封装设计建议

## 设计目标

1. **简化调用** - 隐藏 COM 对象复杂性
2. **统一接口** - 提供一致的 API 风格
3. **错误处理** - 完善的错误捕获和提示
4. **扩展性** - 支持同步/异步/事务等多种模式

## 推荐类结构设计

### 1. 主类 cMSMQClient（消息队列客户端）

```vb
' 类名: cMSMQClient
' 功能: MSMQ 操作主类
'----------------------------------

Private m_QueueInfo As MSMQQueueInfo
Private m_Queue As MSMQQueue
Private m_IsOpen As Boolean
Private m_PathName As String
Private m_Mode As MSMQAccessMode

' 访问模式枚举
Public Enum MSMQAccessMode
    maSendOnly = 0      ' 仅发送
    maReceiveOnly = 1   ' 仅接收
    maSendReceive = 2   ' 发送和接收
    maPeekOnly = 3      ' 仅查看
End Enum

' 队列类型枚举
Public Enum MSMQQueueType
    qtPrivate = 0       ' 专用队列
    qtPublic = 1        ' 公共队列
End Enum

'----------------------------------
' 属性
'----------------------------------
Public Property Get IsOpen() As Boolean
    IsOpen = m_IsOpen
End Property

Public Property Get PathName() As String
    PathName = m_PathName
End Property

Public Property Get MessageCount() As Long
    ' 获取队列中消息数量
    If m_IsOpen Then
        MessageCount = m_QueueInfo.MessageCount
    End If
End Property

'----------------------------------
' 方法: 连接到队列
'----------------------------------
Public Function Connect( _
    ByVal QueueName As String, _
    Optional ByVal QueueType As MSMQQueueType = qtPrivate, _
    Optional ByVal AccessMode As MSMQAccessMode = maSendReceive, _
    Optional ByVal CreateIfNotExist As Boolean = True _
) As Boolean
    ' 实现连接逻辑
End Function

'----------------------------------
' 方法: 发送消息
'----------------------------------
Public Function SendMessage( _
    ByVal MessageBody As Variant, _
    Optional ByVal Label As String = "", _
    Optional ByVal Priority As Long = 3, _
    Optional ByVal IsTransactional As Boolean = False _
) As Boolean
    ' 实现发送逻辑
End Function

'----------------------------------
' 方法: 接收消息（同步）
'----------------------------------
Public Function ReceiveMessage( _
    Optional ByVal Timeout As Long = 0, _
    Optional ByVal WantBody As Boolean = True _
) As cMSMQMessage
    ' 实现接收逻辑，返回 cMSMQMessage 对象
End Function

'----------------------------------
' 方法: 查看消息（不删除）
'----------------------------------
Public Function PeekMessage( _
    Optional ByVal Timeout As Long = 0 _
) As cMSMQMessage
    ' 实现查看逻辑
End Function

'----------------------------------
' 方法: 关闭连接
'----------------------------------
Public Sub Close()
    If Not m_Queue Is Nothing Then
        m_Queue.Close
        Set m_Queue = Nothing
    End If
    Set m_QueueInfo = Nothing
    m_IsOpen = False
End Sub

'----------------------------------
' 清理
'----------------------------------
Private Sub Class_Terminate()
    Close
End Sub
```

### 2. 消息类 cMSMQMessage

```vb
' 类名: cMSMQMessage
' 功能: 封装消息对象
'----------------------------------

Private m_Body As Variant
Private m_Label As String
Private m_Priority As Long
Private m_SentTime As Date
Private m_ArrivedTime As Date
Private m_CorrelationId As String
Private m_Id As String
Private m_Source As MSMQMessage  ' 原始消息对象（内部使用）

'----------------------------------
' 属性
'----------------------------------
Public Property Get Body() As Variant
    Body = m_Body
End Property

Public Property Let Body(ByVal Value As Variant)
    m_Body = Value
End Property

Public Property Get Label() As String
    Label = m_Label
End Property

Public Property Let Label(ByVal Value As String)
    m_Label = Value
End Property

Public Property Get Priority() As Long
    Priority = m_Priority
End Property

Public Property Let Priority(ByVal Value As Long)
    m_Priority = Value
End Property

Public Property Get SentTime() As Date
    SentTime = m_SentTime
End Property

Public Property Get ArrivedTime() As Date
    ArrivedTime = m_ArrivedTime
End Property

Public Property Get MessageId() As String
    MessageId = m_Id
End Property

Public Property Get CorrelationId() As String
    CorrelationId = m_CorrelationId
End Property

'----------------------------------
' 方法: 从 MSMQMessage 对象加载
'----------------------------------
Friend Sub LoadFromMSMQMessage(ByRef SourceMsg As MSMQMessage)
    On Error Resume Next
    m_Body = SourceMsg.Body
    m_Label = SourceMsg.Label
    m_Priority = SourceMsg.Priority
    m_SentTime = SourceMsg.SentTime
    m_ArrivedTime = SourceMsg.ArrivedTime
    m_Id = SourceMsg.Id
    m_CorrelationId = SourceMsg.CorrelationId
    Set m_Source = SourceMsg
End Sub
```

### 4. 事件类 cMSMQEvents（异步接收）

```vb
' 类名: cMSMQEvents
' 功能: 异步消息接收处理
'----------------------------------

Private WithEvents m_Events As MSMQEvent
Attribute m_Events.VB_VarHelpID = -1
Private m_Client As cMSMQClient

'----------------------------------
' 事件声明
'----------------------------------
Public Event OnMessageArrived(ByRef Message As cMSMQMessage)
Public Event OnReceiveError(ByVal ErrorNumber As Long, ByVal ErrorDescription As String)

'----------------------------------
' 方法: 开始监听
'----------------------------------
Public Sub StartListening(ByRef Client As cMSMQClient)
    Set m_Client = Client
    Set m_Events = New MSMQEvent
    ' 启用通知...
End Sub

'----------------------------------
' 方法: 停止监听
'----------------------------------
Public Sub StopListening()
    Set m_Events = Nothing
    Set m_Client = Nothing
End Sub

'----------------------------------
' 事件处理
'----------------------------------
Private Sub m_Events_Arrived(ByVal Queue As Object, ByVal Cursor As Long)
    ' 消息到达处理
    Dim msg As New cMSMQMessage
    ' 加载消息...
    RaiseEvent OnMessageArrived(msg)
End Sub

Private Sub m_Events_ArrivedError(ByVal Queue As Object, ByVal ErrorCode As Long, ByVal Cursor As Long)
    RaiseEvent OnReceiveError(ErrorCode, "MSMQ Receive Error")
End Sub
```

## 使用示例

### 示例1：发送消息
```vb
Dim mq As New cMSMQClient

If mq.Connect("MyQueue", qtPrivate, maSendOnly) Then
    mq.SendMessage "Hello World", "测试消息", 3
    mq.Close
End If
```

### 示例2：接收消息
```vb
Dim mq As New cMSMQClient
Dim msg As cMSMQMessage

If mq.Connect("MyQueue", qtPrivate, maReceiveOnly) Then
    Set msg = mq.ReceiveMessage(5000)  ' 等待5秒
    If Not msg Is Nothing Then
        Debug.Print msg.Body
    End If
    mq.Close
End If
```

### 示例3：异步接收
```vb
Dim WithEvents mqEvents As cMSMQEvents
Dim mq As cMSMQClient

Private Sub Form_Load()
    Set mq = New cMSMQClient
    mq.Connect "MyQueue", qtPrivate, maReceiveOnly
    
    Set mqEvents = New cMSMQEvents
    mqEvents.StartListening mq
End Sub

Private Sub mqEvents_OnMessageArrived(ByRef Message As cMSMQMessage)
    Debug.Print "收到消息: " & Message.Body
End Sub
```
