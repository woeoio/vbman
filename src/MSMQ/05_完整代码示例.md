# VB6 MSMQ 完整代码示例

## 示例1：最简发送/接收

### 发送消息
```vb
Private Sub SendSimpleMessage()
    Dim qinfo As MSMQQueueInfo
    Dim queue As MSMQQueue
    Dim msg As MSMQMessage
    
    On Error GoTo ErrorHandler
    
    ' 创建队列信息对象
    Set qinfo = New MSMQQueueInfo
    qinfo.PathName = ".\private$\TestQueue"
    
    ' 如果队列不存在则创建
    On Error Resume Next
    qinfo.Create
    On Error GoTo ErrorHandler
    
    ' 打开队列（发送模式）
    Set queue = qinfo.Open(MQ_SEND_ACCESS, MQ_DENY_NONE)
    
    ' 创建并发送消息
    Set msg = New MSMQMessage
    msg.Body = "Hello MSMQ!"
    msg.Label = "测试消息"
    msg.Priority = 3
    
    queue.Send msg
    
    MsgBox "消息已发送！"
    
    ' 清理
    queue.Close
    Set queue = Nothing
    Set qinfo = Nothing
    Exit Sub
    
ErrorHandler:
    MsgBox "错误: " & Err.Description, vbCritical
End Sub
```

### 接收消息
```vb
Private Sub ReceiveSimpleMessage()
    Dim qinfo As MSMQQueueInfo
    Dim queue As MSMQQueue
    Dim msg As MSMQMessage
    
    On Error GoTo ErrorHandler
    
    ' 创建队列信息对象
    Set qinfo = New MSMQQueueInfo
    qinfo.PathName = ".\private$\TestQueue"
    
    ' 打开队列（接收模式）
    Set queue = qinfo.Open(MQ_RECEIVE_ACCESS, MQ_DENY_NONE)
    
    ' 接收消息（阻塞模式，超时5秒）
    Set msg = queue.Receive(5000)  ' 5000毫秒 = 5秒
    
    If Not msg Is Nothing Then
        MsgBox "收到消息: " & msg.Body & vbCrLf & _
               "标签: " & msg.Label
    Else
        MsgBox "超时，没有收到消息"
    End If
    
    ' 清理
    queue.Close
    Set queue = Nothing
    Set qinfo = Nothing
    Exit Sub
    
ErrorHandler:
    MsgBox "错误: " & Err.Description, vbCritical
End Sub
```

```vb
Public Type MyMessageData
```

### 发送结构体
```vb
Private Sub SendObjectMessage()
    Dim qinfo As MSMQQueueInfo
    Dim queue As MSMQQueue
    Dim msg As MSMQMessage
    Dim data As MyMessageData
    
    Set qinfo = New MSMQQueueInfo
    qinfo.PathName = ".\private$\ObjectQueue"
    On Error Resume Next: qinfo.Create: On Error GoTo 0
    Set queue = qinfo.Open(MQ_SEND_ACCESS, MQ_DENY_NONE)
    
    ' 填充数据
    data.UserId = 123
    data.UserName = "张三"
    data.Action = "登录"
    data.Timestamp = Now
    
    ' 发送（VB会自动序列化）
    Set msg = New MSMQMessage
    msg.Body = data
    msg.Label = "用户操作"
    msg.AppSpecific = 1  ' 自定义标识，用于区分消息类型
    
    queue.Send msg
    queue.Close
    
    MsgBox "对象已发送！"
End Sub
```

### 接收结构体
```vb
Private Sub ReceiveObjectMessage()
```

## 示例3：事务性消息

### 发送事务消息
```vb
Private Sub SendTransactionalMessage()
    Dim qinfo As MSMQQueueInfo
    Dim queue As MSMQQueue
    Dim msg As MSMQMessage
    Dim txn As MSMQTransaction
    Dim txnDisp As MSMQTransactionDispenser
    
    On Error GoTo ErrorHandler
    
    Set qinfo = New MSMQQueueInfo
    qinfo.PathName = ".\private$\TxnQueue"
    
    ' 创建事务性队列（如果不存在）
    On Error Resume Next
    qinfo.Create True  ' True = 事务性队列
    On Error GoTo ErrorHandler
    
    Set queue = qinfo.Open(MQ_SEND_ACCESS, MQ_DENY_NONE)
    
    ' 创建事务
    Set txnDisp = New MSMQTransactionDispenser
    Set txn = txnDisp.BeginTransaction
    
    ' 发送第一条消息
    Set msg = New MSMQMessage
    msg.Body = "事务消息1"
    queue.Send msg, txn
    
    ' 发送第二条消息
    Set msg = New MSMQMessage
    msg.Body = "事务消息2"
    queue.Send msg, txn
    
    ' 提交事务（两条消息要么都成功，要么都失败）
    txn.Commit
    
    MsgBox "事务提交成功！"
    
    queue.Close
    Exit Sub
    
ErrorHandler:
    MsgBox "错误: " & Err.Description
    ' 回滚事务
    If Not txn Is Nothing Then txn.Abort
    If Not queue Is Nothing Then queue.Close
End Sub
```

## 示例4：异步接收（事件驱动）

### 带事件的窗体代码
```vb
' 窗体代码文件
Option Explicit

Private WithEvents m_Event As MSMQEvent
Attribute m_Event.VB_VarHelpID = -1
Private m_Queue As MSMQQueue

Private Sub Form_Load()
    Dim qinfo As MSMQQueueInfo
    
    ' 打开队列
    Set qinfo = New MSMQQueueInfo
    qinfo.PathName = ".\private$\AsyncQueue"
    On Error Resume Next: qinfo.Create: On Error GoTo 0
    
    Set m_Queue = qinfo.Open(MQ_RECEIVE_ACCESS, MQ_DENY_NONE)
    Set m_Event = New MSMQEvent
    
    ' 启用异步通知
    m_Queue.EnableNotification m_Event
    
    ListBox1.AddItem "开始监听消息..."
End Sub

Private Sub Form_Unload(Cancel As Integer)
    ' 停止监听
    If Not m_Queue Is Nothing Then m_Queue.Close
    Set m_Queue = Nothing
    Set m_Event = Nothing
End Sub

' 消息到达事件处理
Private Sub m_Event_Arrived(ByVal Queue As Object, ByVal Cursor As Long)
    Dim msg As MSMQMessage
    
    On Error Resume Next
    
    ' 接收消息
    Set msg = Queue.Receive(MQ_TRANSACTIONAL_NONE, False, True, 0)
    
    If Not msg Is Nothing Then
        ListBox1.AddItem Format(Now, "HH:MM:SS") & " - " & msg.Body
    End If
    
    ' 继续监听下一条消息
    Queue.EnableNotification m_Event
End Sub

' 错误处理
Private Sub m_Event_ArrivedError(ByVal Queue As Object, ByVal ErrorCode As Long, ByVal Cursor As Long)
    ListBox1.AddItem "错误: " & ErrorCode
    ' 继续监听    Queue.EnableNotification m_Event
End Sub
```

## 示例5：批量发送和接收

### 批量发送
```vb
Private Sub BatchSendMessages()
    Dim qinfo As MSMQQueueInfo
    Dim queue As MSMQQueue
    Dim msg As MSMQMessage
    Dim i As Long
    
    Set qinfo = New MSMQQueueInfo
    qinfo.PathName = ".\private$\BatchQueue"
    On Error Resume Next: qinfo.Create: On Error GoTo 0
    Set queue = qinfo.Open(MQ_SEND_ACCESS, MQ_DENY_NONE)
    
    ' 批量发送100条消息
    For i = 1 To 100
        Set msg = New MSMQMessage
        msg.Body = "消息 #" & i
        msg.Label = "批量消息"
        msg.AppSpecific = i  ' 用 AppSpecific 存储序号
        
        queue.Send msg
        
        ' 每10条更新一次UI
        If i Mod 10 = 0 Then
            Label1.Caption = "已发送: " & i & "/100"
            DoEvents
        End If
    Next i
    
    queue.Close
    MsgBox "批量发送完成！"
End Sub
```

### 批量接收
```vb
Private Sub BatchReceiveMessages()
    Dim qinfo As MSMQQueueInfo
    Dim queue As MSMQQueue
    Dim msg As MSMQMessage
    Dim count As Long
    
    Set qinfo = New MSMQQueueInfo
    qinfo.PathName = ".\private$\BatchQueue"
    Set queue = qinfo.Open(MQ_RECEIVE_ACCESS, MQ_DENY_NONE)
    
    count = 0
    
    ' 循环接收所有消息
    Do
        Set msg = queue.Receive(1000)  ' 1秒超时
        
        If msg Is Nothing Then
            Exit Do  ' 没有更多消息
        End If
        
        count = count + 1
        ListBox1.AddItem "#" & msg.AppSpecific & ": " & msg.Body
        
        ' 每接收10条更新UI
        If count Mod 10 = 0 Then
            Label1.Caption = "已接收: " & count
            DoEvents
        End If
    Loop
    
    queue.Close
    MsgBox "共接收 " & count & " 条消息"
End Sub
```
