'发送聊天请求
'Stream=False 时同步返回完整结果
'Stream=True 时通过 SSE 流式接收，阻塞等待直到完成
Public Function Chat(Optional UserMessage As String, Optional Callback As Object, Optional CallbackMethod As String) As String
    On Error GoTo ERR_HANDLER
    
    '添加用户消息（如果提供）
    If UserMessage <> "" Then
        Call Me.User(UserMessage)
    End If
    
    '验证配置
    If m_ApiKey = "" Then
        ERR.Raise 500, , "API Key 未设置，请使用 ApiKey() 或预设配置方法设置"
    End If
    
    '设置默认值
    If m_BaseUrl = "" Then m_BaseUrl = "https://api.openai.com/v1"
    If m_ApiPath = "" Then m_ApiPath = "/chat/completions"
    If m_Model = "" Then m_Model = "gpt-3.5-turbo"
    If m_Timeout = 0 Then m_Timeout = 60
    
    '检查是否启用流式响应
    If m_Stream = True Then
        '检查是否已有流式请求在执行
        If m_IsStreaming Then
            ERR.Raise 500, , "上一次流式请求尚未完成"
        End If
        
        m_StreamCallback = Callback
        m_StreamCallbackMethod = CallbackMethod
        m_StreamBuffer = ""
        
        Call SendStreamRequest
        
        '等待流式响应完成（阻塞等待）
        Do While m_IsStreaming
            DoEvents
        Loop
        
        Chat = m_StreamBuffer
    Else
        '同步请求
        Chat = SendSyncRequest
    End If
    
    Exit Function
    
ERR_HANDLER:
    m_LastError = ERR.Description
    RaiseEvent OnError(ERR.Description, ERR.Number)
    ERR.Raise ERR.Number, ERR.Source, ERR.Description
End Function

'构建请求体（不修改 m_Messages，避免系统提示词重复添加）
Private Function BuildRequestBody() As String
    Set Json = New cJson
    
    '设置模型
    Json("model") = m_Model
    
    '构建消息列表（使用临时集合，不修改 m_Messages）
    Dim MsgList As New Collection
    
    '如果有系统提示词，添加到消息列表开头
    If m_SystemPrompt <> "" Then
        Dim SysMsg As New Scripting.Dictionary
        SysMsg("role") = "system"
        SysMsg("content") = m_SystemPrompt
        MsgList.Add SysMsg
    End If
    
    '添加用户/助手消息
    If Not m_Messages Is Nothing Then
        Dim i As Long
        For i = 1 To m_Messages.Count
            MsgList.Add m_Messages(i)
        Next
    End If
    
    '设置消息
    Json("messages") = MsgList
    
    '设置参数
    If m_Temperature <> 1# Then
        Json("temperature") = m_Temperature
    End If
    
    If m_MaxTokens > 0 Then
        Json("max_tokens") = m_MaxTokens
    End If
    
    If m_TopP <> 1# Then
        Json("top_p") = m_TopP
    End If
    
    If m_FrequencyPenalty <> 0 Then
        Json("frequency_penalty") = m_FrequencyPenalty
    End If
    
    If m_PresencePenalty <> 0 Then
        Json("presence_penalty") = m_PresencePenalty
    End If
    
    If m_Stream = True Then
        Json("stream") = True
    End If
    
    If Not IsEmpty(m_Stop) Then
        Json("stop") = m_Stop
    End If
    
    BuildRequestBody = Json.Encode()
End Function

Public Function Start(Optional Port As Long = 80, Optional WebRoot As String = "", Optional IP As String = "0.0.0.0") As Boolean
    If IsRunning = True Then LastError = "服务器正在运行中": Exit Function
    Set m_oServer = New cTlsReMaster
    With m_oServer
        .Protocol = RemasterProtocolConstants.sckTCPProtocol
        .Bind Port, IP
        .Listen
    End With
    '    Shell "cmd /c start http://localhost:8088/"
    Set HttpApp = New cHttpServerApp
    Set HttpSvr = New cHttpServerSvr
    With HttpSvr
        .IP = IP
        .Port = Port
        .WebRoot = WebRoot
    End With
    IsRunning = True
    Start = True
End Function

Public Function StopMe() As Boolean
    On Error GoTo EH
    m_oServer.CloseSck
    IsRunning = False
    ' 逐一释放所有连接，断开循环引用
    Dim i As Long
    For i = m_cConnPool.Count To 1 Step -1
        Dim oCallback As cClientCallback
        Set oCallback = m_cConnPool.Item(i)
        If Not oCallback Is Nothing Then
            oCallback.Release
        End If
    Next
    Set m_cConnPool = Nothing
    Set HttpApp = Nothing
    Set HttpSvr = Nothing
    Set m_oServer = Nothing
    
    Set SSE = Nothing
    Set ScriptEngine = Nothing
    
    Set Router = Nothing
    Set RouteBefore = Nothing
    Set RouteAfter = Nothing
    Set Database = Nothing
    
    Set CrossDomain = Nothing
    
    StopMe = True
    Exit Function
EH:
    LastError = ERR.Number & "#" & ERR.Description
End Function

Friend Sub OnDataArrival(Index As Long)
    
    On Error GoTo EH
    Dim oCallback As cClientCallback
    Set oCallback = m_cConnPool.Item(Index)
    
    ' 1. 获取新数据并追加到接收缓冲区（解决粘包问题）
    Dim newData() As Byte
    oCallback.Socket.GetData newData, vbArray + vbByte
    
    ' 追加数据到缓冲区
    oCallback.AppendToBuffer newData
    
    ' 2. 循环处理缓冲区中的完整请求（解决粘包问题）
    Do While ContainsCompleteRequest(oCallback)
        ' 3. 提取一个完整的 HTTP 请求
        Dim completeRequest As Variant
        completeRequest = ExtractOneRequest(oCallback)
        
        ' 4. 处理该请求
        ProcessHttpRequest Index, completeRequest
        
        ' 5. 从缓冲区移除已处理的数据
        RemoveProcessedData oCallback, completeRequest
    Loop
    Exit Sub
EH:
    Dim ErrInfo As String: ErrInfo = ERR.Number & "#" & ERR.Description
    RaiseEvent OnLogs(ToolsLogs.LEVEL_ERROR, "OnDataArrival: " & ErrInfo)
    ' 发生错误时清空缓冲区，避免一直累积错误数据
    oCallback.GetRecvBufferLen = 0
End Sub

' 判断是否包含完整的 HTTP 请求（用于解决粘包问题）
Private Function ContainsCompleteRequest(oCallback As cClientCallback) As Boolean
    Dim bufferLen As Long
    bufferLen = oCallback.GetRecvBufferLen
    
    ' 如果缓冲区为空，直接返回 False
    If bufferLen = 0 Then
        ContainsCompleteRequest = False
        Exit Function
    End If
    
    ' 获取缓冲区数据
    Dim buffer() As Byte
    buffer = oCallback.GetRecvBuffer
    
    ' 查找请求头结束标记 CRLFCRLF
    Dim headerEnd As Long
    headerEnd = FindCRLFCRLF(buffer, bufferLen)
    
    If headerEnd = 0 Then
        ' 未找到完整的请求头
        ContainsCompleteRequest = False
        Exit Function
    End If
    
    ' 将请求头部分转换为字符串
    Dim headerBytes() As Byte
    ReDim headerBytes(0 To headerEnd - 1)
    CopyMemory headerBytes(0), buffer(0), headerEnd
    
    Dim headerStr As String
    headerStr = ToolsUtf8.Decode(headerBytes)
    
    ' 解析 Content-Length
    Dim contentLength As Long
    contentLength = ParseContentLengthFromHeader(headerStr)
    
    ' 计算已接收的体部长度
    Dim bodyStart As Long
    bodyStart = headerEnd + 4  ' \r\n\r\n 的长度（4个字节）
    
    ' 如果是 GET、HEAD、OPTIONS 等没有 body 的请求，contentLength 为 0
    If contentLength = 0 Then
        ' 检查是否是这些没有 body 的请求方法
        If IsMethodWithoutBody(headerStr) Then
            ContainsCompleteRequest = True
            Exit Function
        End If
    End If
    
    ' 判断缓冲区是否包含完整的请求（头部 + 体部）
    Dim totalRequired As Long
    totalRequired = bodyStart + contentLength
    
    ContainsCompleteRequest = (bufferLen >= totalRequired)
End Function

' 查找 CRLFCRLF 标记
Private Function FindCRLFCRLF(buffer() As Byte, bufLen As Long) As Long
    Dim i As Long
    For i = 0 To bufLen - 4
        If buffer(i) = 13 And buffer(i + 1) = 10 And _
           buffer(i + 2) = 13 And buffer(i + 3) = 10 Then
            FindCRLFCRLF = i
            Exit Function
        End If
    Next
    FindCRLFCRLF = 0
End Function

Private Function ExtractOneRequest(oCallback As cClientCallback) As Variant

    bufferLen = oCallback.GetRecvBufferLen

    Dim bodyStart As Long
    bodyStart = headerEnd + 4

        ReDim bodyBytes(0 To contentLength - 1)
        CopyMemory bodyBytes(0), buffer(bodyStart), contentLength
        result(1) = bodyBytes
    Else
        ' 没有 body
        result(1) = vbNullString
    End If
    
    ' 返回总长度
    result(2) = bodyStart + contentLength

' 从请求头字符串中解析 Content-Length
Private Function ParseContentLengthFromHeader(headerStr As String) As Long
    On Error GoTo EH
    Dim lines As Variant
    lines = Split(headerStr, vbCrLf)
    
    Dim i As Long
    For i = 0 To UBound(lines)
        Dim line As String
        line = lines(i)
        
        ' 查找 Content-Length 头（不区分大小写）
        If InStr(1, line, "Content-Length:", vbTextCompare) > 0 Then
            Dim parts As Variant
            parts = Split(line, ":", 2)
            If UBound(parts) >= 1 Then
                ParseContentLengthFromHeader = CLng(Trim(parts(1)))
                Exit Function
            End If
        End If
    Next
    
    ' 如果没有找到 Content-Length，返回 0
    ParseContentLengthFromHeader = 0
    Exit Function
EH:
    ParseContentLengthFromHeader = 0
End Function

' 判断请求方法是否没有 Body
Private Function IsMethodWithoutBody(headerStr As String) As Boolean
    On Error GoTo EH
    ' 获取第一行
    Dim firstLine As String
    Dim lines As Variant
    lines = Split(headerStr, vbCrLf)
    If UBound(lines) >= 0 Then
        firstLine = lines(0)
        
        ' 检查请求方法
        If InStr(1, firstLine, "GET ", vbTextCompare) = 1 Or _
           InStr(1, firstLine, "HEAD ", vbTextCompare) = 1 Or _
           InStr(1, firstLine, "OPTIONS ", vbTextCompare) = 1 Or _
           InStr(1, firstLine, "DELETE ", vbTextCompare) = 1 Then
            IsMethodWithoutBody = True
            Exit Function
        End If
    End If
    IsMethodWithoutBody = False
    Exit Function
EH:
    IsMethodWithoutBody = False
End Function

Public Function Decode(ByRef Source As Variant) As cJson
    ' [using]: ToolsJsonVba.ParseJson from
    ' VBA-JSON v2.3.1
    ' (c) Tim Hall - https://github.com/VBA-tools/VBA-JSON
    Set Decode = Me
    On Error GoTo EH
    LastError = ""
    
    ' 支持多种类型转换（类似 Encode）
    If IsObject(Source) = True Then
        If TypeOf Source Is ADODB.Recordset Then
            Set RootItems = ToolsList.RsToCollection(Source).RawCollection
            Exit Function
        ElseIf TypeOf Source Is cCollection Then
            If TypeOf Source.RawCollection Is Collection Then
                Set RootItems = Source.RawCollection
            Else
                Set RootItem = Source.RawCollection
            End If
            Exit Function
        ElseIf TypeOf Source Is cJson Then
            Set RootItem = Source.RootItem
            Set RootItems = Source.RootItems
            RootText = Source.RootText
            Exit Function
        ElseIf TypeOf Source Is Collection Then
            Set RootItems = Source
            Exit Function
        ElseIf TypeOf Source Is Scripting.Dictionary Then
            Set RootItem = Source
            Exit Function
        End If
    End If
    
    RootText = ToolsStr.TrimEx(CStr(Source))
Start:
    If RootText = "" Then ERR.Raise 1, "", "No JSON data."
    If Left(RootText, 1) = "[" Then
        Set RootItems = ParseJson(RootText)
    ElseIf Left(RootText, 1) = "{" Then
        Set RootItem = ParseJson(RootText)
    Else
        ' check if JSONP / Var
        GoSub CheckJSONP
        GoSub CheckVar
        'error data
        ERR.Raise 1, "", "Not stand JSON data."
    End If
    Exit Function
CheckJSONP:
    Dim LeftK As Long: LeftK = ToolsStr.HasStr("(", RootText)
    Dim RightK As Long: RightK = ToolsStr.HasStrFromRight(");", RootText & ";")
    If LeftK > 0 And RightK > 0 Then
        Dim LastChar As String: LastChar = Right(RootText, 1)
        If LastChar = ";" Then LastChar = ");"
        RootText = ToolsStr.SubStr(RootText, "(", LastChar, False, vbBinaryCompare, True)
        GoTo Start
    End If
    Return
CheckVar:
    Dim LeftWord As String: LeftWord = ToolsStr.GetFirstChar(RootText, 3)
    If LeftWord = "var" Then
        RootText = ToolsStr.SubStr(RootText, "=")
        GoTo Start
    End If
    Return
EH:
    ERR.Raise ERR.Number, "DECODE", ERR.Description & "#" & CStr(Source)
End Function

' Broadcast binary message to all connected clients
Public Sub BroadcastBinary(Data() As Byte, Optional ByVal ExcludeClientID As String = "")
    On Error Resume Next

    Dim oClient As cWebSocketServerClient
    Dim baFrame() As Byte

    ' Build frame once
    baFrame = m_FrameBuilder.BuildBinaryFrame(Data, False)

    For Each oClient In m_Clients
        If oClient.State = WS_CLIENT_CONNECTED Then
            If LenB(ExcludeClientID) = 0 Or oClient.ID <> ExcludeClientID Then
                oClient.SendFrame baFrame
            End If
        End If
    Next
End Sub

' Disconnect specific client
Public Sub DisconnectClient(ByVal ClientID As String, _
                           Optional ByVal Code As WsCloseCode = WS_CLOSE_NORMAL, _
                           Optional ByVal Reason As String = "")
    On Error Resume Next

    Dim oClient As cWebSocketServerClient
    Set oClient = GetClient(ClientID)

    If Not oClient Is Nothing Then
        oClient.SendClose Code, Reason
        oClient.CloseSocket
    End If
End Sub

        m_Clients.Remove ClientID

    End If
End Sub

Private Function HandleHandshake(ByVal Client As cWebSocketServerClient, ByVal Request As String) As Boolean
    Dim sKey As String
    Dim sAccept As String
    Dim sResponse As String

    On Error GoTo ErrorHandler

    ' Check for WebSocket upgrade request
    If InStr(1, Request, "Upgrade: websocket", vbTextCompare) = 0 Then
        HandleHandshake = False
        Exit Function
    End If

    ' Get Sec-WebSocket-Key
    sKey = GetHeaderValue(Request, "Sec-WebSocket-Key")
    If LenB(sKey) = 0 Then
        HandleHandshake = False
        Exit Function
    End If

    ' Compute accept key
    sAccept = ComputeAcceptKey(sKey)
    If LenB(sAccept) = 0 Then
        HandleHandshake = False
        Exit Function
    End If

    ' Build response
    sResponse = "HTTP/1.1 101 Switching Protocols" & vbCrLf
    sResponse = sResponse & "Upgrade: websocket" & vbCrLf
    sResponse = sResponse & "Connection: Upgrade" & vbCrLf
    sResponse = sResponse & "Sec-WebSocket-Accept: " & sAccept & vbCrLf
    sResponse = sResponse & vbCrLf

    ' Send response
    Client.Socket.SendData sResponse, wcpUtf8

    HandleHandshake = True
    Exit Function

ErrorHandler:
    HandleHandshake = False
End Function

'=========================================================================
' Private Methods - Frame Processing
'=========================================================================

Private Sub ProcessClientData(ByVal Client As cWebSocketServerClient)
    Dim oParser As cWebSocketFrame
    Dim baPayload() As Byte

    On Error GoTo ErrorHandler

    Set oParser = Client.FrameParser

    Do While Client.RecvBuffer.Size >= 2
        ' Parse frame header
        If Not oParser.ParseHeader(Client.RecvBuffer) Then
            Exit Do  ' Need more data
        End If

        ' Check if complete frame available
        If Not oParser.IsCompleteFrame(Client.RecvBuffer) Then
            Exit Do  ' Need more data
        End If

        ' Extract payload
        baPayload = oParser.ExtractPayload(Client.RecvBuffer)

        ' Process by opcode
        Select Case oParser.OpCode
            Case WS_OPCODE_TEXT, WS_OPCODE_BINARY
                ProcessDataFrame Client, baPayload, oParser.OpCode, oParser.FIN

            Case WS_OPCODE_CONTINUATION
                ProcessContinuationFrame Client, baPayload, oParser.FIN

            Case WS_OPCODE_CLOSE
                ProcessCloseFrame Client, baPayload

            Case WS_OPCODE_PING
                ProcessPingFrame Client, baPayload

            Case WS_OPCODE_PONG
                ' Ignore pong
        End Select
    Loop

    Exit Sub
ErrorHandler:
    RaiseError "ProcessClientData: " & ERR.Description
End Sub

Private Sub ProcessDataFrame(ByVal Client As cWebSocketServerClient, _
                            ByRef Payload() As Byte, _
                            ByVal OpCode As WsOpCode, _
                            ByVal IsFinal As Boolean)
    If IsFinal And Not Client.IsFragmented Then
        ' Complete standalone message
        DeliverClientMessage Client, Payload, OpCode
    Else
        ' Start fragmented message
        Client.IsFragmented = True
        Client.FragmentOpCode = OpCode
        Client.FragmentBuffer.Clear

        On Error Resume Next
        If UBound(Payload) >= 0 Then
            Client.FragmentBuffer.Append Payload
        End If
        On Error GoTo 0

        If IsFinal Then
            DeliverFragmentedMessage Client
        End If
    End If
End Sub

Private Sub ProcessContinuationFrame(ByVal Client As cWebSocketServerClient, _
                                    ByRef Payload() As Byte, _
                                    ByVal IsFinal As Boolean)
    If Not Client.IsFragmented Then
        RaiseError "Unexpected continuation frame from " & Client.ID
        Exit Sub
    End If

    On Error Resume Next
    If UBound(Payload) >= 0 Then
        Client.FragmentBuffer.Append Payload
    End If
    On Error GoTo 0

    If IsFinal Then
        DeliverFragmentedMessage Client
    End If
End Sub

Private Sub DeliverFragmentedMessage(ByVal Client As cWebSocketServerClient)
    Dim baData() As Byte
    baData = Client.GetFragmentedData

    DeliverClientMessage Client, baData, Client.FragmentOpCode

    Client.ClearFragmentBuffer
End Sub

Private Sub DeliverClientMessage(ByVal Client As cWebSocketServerClient, _
                                ByRef Data() As Byte, _
                                ByVal OpCode As WsOpCode)
    On Error Resume Next

    If OpCode = WS_OPCODE_TEXT Then
        Dim sText As String
        If UBound(Data) >= 0 Then
            sText = UTF8ToString(Data)
        Else
            sText = ""
        End If
        RaiseEvent OnClientTextMessage(Client.ID, sText)
    Else
        RaiseEvent OnClientBinaryMessage(Client.ID, Data)
    End If
End Sub

Private Sub ProcessCloseFrame(ByVal Client As cWebSocketServerClient, ByRef Payload() As Byte)
    Dim sReason As String

    On Error Resume Next
    sReason = "Client initiated close"

    ' Parse reason if available
    If UBound(Payload) >= 3 Then
        Dim baReason() As Byte
        ReDim baReason(UBound(Payload) - 2) As Byte
        Dim i As Long
        For i = 2 To UBound(Payload)
            baReason(i - 2) = Payload(i)
        Next i
        sReason = UTF8ToString(baReason)
    End If

    On Error GoTo 0

    RemoveClient Client.ID, sReason
End Sub

Private Sub ProcessPingFrame(ByVal Client As cWebSocketServerClient, ByRef Payload() As Byte)
    ' Auto-reply with pong
    Client.SendPong Payload
End Sub

Private Sub m_ListenSocket_ConnectionRequest(Client As cWinsock, ByRef DisConnect As Boolean)
    On Error GoTo ErrorHandler

    ' Create new client
    Dim oClient As cWebSocketServerClient
    Set oClient = New cWebSocketServerClient

    ' Generate client ID
    m_ClientCounter = m_ClientCounter + 1
    oClient.ID = "Client#" & m_ClientCounter

    ' Set client properties
    Set oClient.Socket = Client
    oClient.RemoteAddress = Client.RemoteHostIP
    oClient.RemotePort = Client.RemotePort
    oClient.State = WS_CLIENT_PENDING

    ' Add to collection
    m_Clients.Add oClient, oClient.ID

    ' Accept connection
    DisConnect = False

    Exit Sub
ErrorHandler:
    DisConnect = True
    RaiseError "ConnectionRequest: " & ERR.Description
End Sub

Private Sub m_ListenSocket_DataArrival(Client As cWinsock, ByVal bytesTotal As Long)
    Dim oClient As cWebSocketServerClient
    Dim baData() As Byte
    Dim sData As String

    On Error GoTo ErrorHandler

    ' Find client
    Set oClient = FindClientBySocket(Client)
    If oClient Is Nothing Then Exit Sub

    If oClient.State = WS_CLIENT_PENDING Then
        ' Expecting handshake
        Client.GetData sData, vbString, , wcpUtf8

        If HandleHandshake(oClient, sData) Then
            oClient.State = WS_CLIENT_CONNECTED
            RaiseEvent OnClientConnect(oClient.ID, oClient.RemoteAddress, oClient.RemotePort)
        Else
            ' Handshake failed
            Client.Close_
            m_Clients.Remove oClient.ID
            RaiseError "Handshake failed for " & oClient.ID
        End If

    ElseIf oClient.State = WS_CLIENT_CONNECTED Then
        ' Receive WebSocket data
        Client.GetData baData, vbByte + vbArray

        On Error Resume Next
        If UBound(baData) >= 0 Then
            oClient.AppendRecvData baData
        End If
        On Error GoTo ErrorHandler

        ProcessClientData oClient
    End If

    Exit Sub
ErrorHandler:
    RaiseError "DataArrival: " & ERR.Description
End Sub

Private Sub m_ListenSocket_CloseEvent(Client As cWinsock)
    Dim oClient As cWebSocketServerClient

    On Error Resume Next

    Set oClient = FindClientBySocket(Client)
    If Not oClient Is Nothing Then
        RemoveClient oClient.ID, "Connection closed"
    End If
End Sub

Private Sub m_ListenSocket_Error(Client As cWinsock, ByVal Number As Long, Description As String, ByVal Scode As Long)
    Dim oClient As cWebSocketServerClient

    On Error Resume Next

    Set oClient = FindClientBySocket(Client)
    If Not oClient Is Nothing Then
        RemoveClient oClient.ID, "Socket error: " & Description
    Else
        RaiseError "Socket error: " & Description
    End If
End Sub
