'计算字节数组哈希
Public Function HashBytes(ByRef Data() As Byte, ByVal Algorithm As HashAlgorithm) As String
    On Error GoTo EH
    
    Dim hProv As Long
    Dim hHash As Long
    Dim Algid As Long
    Dim HashLen As Long
    Dim HashData() As Byte
    Dim i As Long
    Dim Result As String
    
    m_LastError = ""
    
    '选择算法
    Select Case Algorithm
        Case HashMD5: Algid = CALG_MD5: HashLen = 16
        Case HashSHA1: Algid = CALG_SHA1: HashLen = 20
        Case HashSHA256: Algid = CALG_SHA256: HashLen = 32
        Case Else
            m_LastError = "不支持的算法"
            Exit Function
    End Select
    
    '获取CSP上下文
    If CryptAcquireContext(hProv, vbNullString, vbNullString, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT) = 0 Then
        m_LastError = "获取加密上下文失败"
        Exit Function
    End If
    
    '创建哈希对象
    If CryptCreateHash(hProv, Algid, 0, 0, hHash) = 0 Then
        CryptReleaseContext hProv, 0
        m_LastError = "创建哈希对象失败"
        Exit Function
    End If
    
    '哈希数据
    If CryptHashData(hHash, Data(0), UBound(Data) + 1, 0) = 0 Then
        CryptDestroyHash hHash
        CryptReleaseContext hProv, 0
        m_LastError = "哈希数据失败"
        Exit Function
    End If
    
    '获取哈希值
    ReDim HashData(HashLen - 1)
    If CryptGetHashParam(hHash, HP_HASHVAL, HashData(0), HashLen, 0) = 0 Then
        CryptDestroyHash hHash
        CryptReleaseContext hProv, 0
        m_LastError = "获取哈希值失败"
        Exit Function
    End If
    
    '转换为十六进制字符串
    Result = ""
    For i = 0 To HashLen - 1
        Result = Result & Right("0" & Hex(HashData(i)), 2)
    Next
    
    '清理
    CryptDestroyHash hHash
    CryptReleaseContext hProv, 0
    
    HashBytes = LCase(Result)
    Exit Function
    
EH:
    m_LastError = ERR.Number & "#" & ERR.Description
    HashBytes = ""
End Function

    Dim temp As Variant
    
    Result = arr
    
    For i = LBound(Result) To UBound(Result) - 1
        For j = i + 1 To UBound(Result)
            If Ascending Then
                If Result(i) > Result(j) Then
                    temp = Result(i)
                    Result(i) = Result(j)
                    Result(j) = temp
                End If
            Else
                If Result(i) < Result(j) Then
                    temp = Result(i)
                    Result(i) = Result(j)
                    Result(j) = temp
                End If
            End If
        Next
    Next
    
    Sort = Result
    Exit Function
    
EH:
    Sort = arr
End Function

Public Function Reverse(ByRef arr As Variant) As Variant
    On Error GoTo EH
    Dim Result() As Variant
    Dim i As Long, j As Long
    
    ReDim Result(UBound(arr))
    j = UBound(arr)
    For i = LBound(arr) To UBound(arr)
        Result(j) = arr(i)
        j = j - 1
    Next
    
    Reverse = Result
    Exit Function
    
EH:
    Reverse = arr
End Function

'URL编码
Public Function UrlEncode(ByVal Text As String) As String
    Dim i As Long
    Dim c As String
    Dim Result As String
    Result = ""
    For i = 1 To Len(Text)
        c = Mid(Text, i, 1)
        Select Case c
            Case "A" To "Z", "a" To "z", "0" To "9", "-", "_", ".", "~"
                Result = Result & c
            Case " "
                Result = Result & "+"
            Case Else
                Result = Result & "%" & Right("0" & Hex(Asc(c)), 2)
        End Select
    Next
    UrlEncode = Result
End Function

'URL解码
Public Function UrlDecode(ByVal Text As String) As String
    Dim i As Long
    Dim c As String
    Dim Result As String
    Result = ""
    i = 1
    Do While i <= Len(Text)
        c = Mid(Text, i, 1)
        If c = "+" Then
            Result = Result & " "
        ElseIf c = "%" And i + 2 <= Len(Text) Then
            Result = Result & Chr(CInt("&H" & Mid(Text, i + 1, 2)))
            i = i + 2
        Else
            Result = Result & c
        End If
        i = i + 1
    Loop
    UrlDecode = Result
End Function

Public Function HtmlEncode(ByVal Text As String) As String
    Dim Result As String
    Result = Text

    Result = Replace(Result, "<", "&lt;")
    Result = Replace(Result, ">", "&gt;")
    Result = Replace(Result, """", "&quot;")
    HtmlEncode = Result
End Function
